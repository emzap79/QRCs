% http://www.linuxjournal.com/article/10636
\section{Movements}{}
\cmdS{hjkl}	{move left/down/up/right}
\cmdS{gg~G}	{move to top/bottom of page}
\cmdS{/~?}	{search in page (backwards)}
\cmdS{\space~\shift\space}	{move down/up one page}

\section{Windows \& Tabs}{}
\cmdS{gt~\ctrl n}	{go to next tab}
\cmdS{gT~\ctrl p}	{go to previous tab}
\cmdS{gh~gH}	{go to your homepage (in new tab)}
\cmdS{gu~gU}	{move up one path/all to root}
\cmdExmpl{http://www.reddit.com/r/IAmA/comments moves up to www.redd\ldots /r/IAmA}	{gu}
\cmdS{t~T}	{open tab (with smart completion)}
\cmdS{d~u}	{close/undo tab}
\cmdS{H~L}	{go to previous/next page in history}
\cmdS{i~I}	{ignores all/specific VIMPRTR keybinds}
\cmdS{o~O}	{}
\cmdS{zi~zI}	{zoom in by 25/100 \% on page in focus}
\cmdS{zo~zO}	{zoom out by 25 /100 \% on focused page}
\cmdS{$n$zz}	{zoom page to $n$\% (100 if $n$ is omitted)}
\cmdS{ZZ~ZQ}	{exit Firefox and save/clear session history}

\section{Modes}{exit mode with \key{esc}. To open link in new tab change to uppercase, eg. $F$ instead of $f$.}
\cmdS{a~A$n$}	{open link $n$ (in new background tab)}
\cmdS{y~Y}	{copy URL/label description to clipboard}
\cmdS{s~S}	{save destination of a link}
\subsection{QuickHint Mode ($f$)}{add numbered label to every link on webpage}
\subsection{ExtendedHint ( ; )}{lets you copy labeled links to clipboard}

\section{Exe commands}{}
\cmdS{:history}	{pull up most recent 10 online \& local pages}
\cmdOper{;o}	{follow hint in history}
\cmdOper{\space~d~j}	{move screen/page/line down}
\cmdOper{b~u~k}	{move screen/page/line up}
\cmdOper{g~G}	{jump to top/bottom of buffer}
\cmdOper{esc~q}	{close buffer}
\cmdS{:pc\grey{lose}}	{close preview window, eg. history}
\cmdL{:bm\grey{ark}~:delbm\grey{marks} $url$}	{add/delete bookmark}
\cmdS{:his\grey{tory}}	{open history}
\cmdS{:res\grey{tart}}	{restart Firefox immediately}
