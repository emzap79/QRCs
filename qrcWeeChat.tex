\section{Basic Operations}{}
\cmdS{\bs help \key{cmd}}	{To get help on a specific command}
\cmdS{\bs set}	{display all options (WeeChat and plugins)}
\cmdL{\bs set weechat.*}	{display WeeChat options}
\cmdS{\bs set irc.*}	{display IRC plugin options}

\subsection{Movements}{moving around in weechat's command line is pretty much the same as in terminal (see: qrcShell)}

\subsection{Buffer}{}
\cmdS{\key{\ctrl s \ctrl u}}	{Set unread marker on all windows}
\cmdS{\ctrl p \ctrl n}	{switch buffer left, right}
\cmdS{\alt a}	{goto next active buffer}
\cmdS{\alt 0\ldots9}	{switch to buffer $n$}
\cmdS{F9/F10}	{scroll buffer title}
\cmdS{F11/F12}	{scroll nick list}

\cmdS{\alt w \alt <}	{switch windows}
\cmdS{\alt w \alt b}	{balance windows}

\subsection{Commands}{}
\cmdS{/window zoom}	{zoom into window}
\cmdL{/window splith, splitv}	{split window horizontally, vertically}
\cmdL{/join \#channel}	{join a channel}
\cmdL{/part [quit message]}	{part a channel, keeping buffer open}
\cmdL{/query $nick$ $message$}	{send message to $nick$}
\cmdS{/close}	{close server or (private) channel buffer}
\cmdS{/close}	{close server or channel buffer}
