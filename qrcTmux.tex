\section{From Shell}{to reach the command mode, use \ctrl b:}

\cmdM{create session}	{new -s {\tt $foo$}}
\cmdM{kill session}	{kill-session -t {\tt $foo$}}
\cmdM{attach session}	{a\grey{ttach} -t {\tt $foo$}\or\ a \#}
\cmdM{rename session}	{rename-session -t $old$ $new$}
\cmdM{list session}	{ls}
\cmdM{move session}	{move-window -t $other\_session$}

\section{Sessions}{\metakeydef}
\cmdS{\ctrl b?}	{show all available keybinds}
\cmdS{\ctrl bL}	{switch back to previous session}
\cmdS{\ctrl bd \ctrl bs}	{detach from,  list all sessions}

\section{Panes and Buffers}{}
\subsection{Copy Mode}{{\tt vim} like movements}
\cmdS{\ctrl b\lbracket ~ q}	{enter, exit copy mode}
\cmdS{h j k l}{move left, down, up, right}
\cmdS{\key{space} \enter}	{select,  copy text under cursor}
\cmdS{\ctrl b\rbracket}	{paste copied text to command line}

\subsection{Open, Switch, Delete}{}
\cmdS{\ctrl b$n$}	{switch to buffer $0 \ldots n$}
\cmdS{\ctrl bc~\ctrl b,~\ctrl bx}	{create, rename, delete window}
\cmdS{\ctrl bp \ctrl bn \ctrl bl}	{go to prev, next, last window}
\cmdS{\ctrl b\arrows}	{switch to left, right, top, bottom pane}
\cmdS{\ctrl bf~\ctrl bw}	{find, list windows}
\cmdS{\ctrl bt \ctrl bq}	{show time, numeric values of panes}
\cmdS{\ctrl b\ctrl o}	{swap with previous pane}
\cmdS{\ctrl b$\lbrace$ \ctrl b$\rbrace$}	{swap position with previous, next pane}
\cmdS{\ctrl b\% \ctrl b"}	{split pane vertically, horizontally}
\cmdS{\ctrl b\key{M - \arrows }}	{resize pane left, right, up, down}
\cmdS{\ctrl b!}	{push all other panes to next window}

\subsection{Panes Position}{}
\cmdS{\ctrl b\key{space}}	{toggle through all layouts}
\cmdS{\ctrl b\key{M-1}}	{vertical split, all panes same width}
\cmdS{\ctrl b\key{M-2}}	{horizontal split, all panes same height}
\cmdS{\ctrl b\key{M-3}}	{horizontal split, main pane on top, other panes on bottom, vertically split, all same width}
\cmdS{\ctrl b\key{M-4}}	{vertical split, main pane left, other panes right, horizontally split, all same height}
\cmdS{\ctrl b\key{M-5}}	{tile, new panes on bottom, same height before same width}
