\def\cmdSB#1#2{
  \noindent
  \hbox to \hsize {%
    \vtop{
      \hbox to 2.4cm {
        \noindent\cmd{#1}\dotfill
      }
      %\hrule
    }%
    \hfil	% absorb extra space b/w columns
    \vtop{
      \hsize=5.90cm
      \hbox{\hfuzz = 15pt \vtop{
      {#2}
      }}
      %\hrule
    }
  }
  %\hrule
  \par
  \vskip 0.14cm
}

%}}}
