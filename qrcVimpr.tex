\section{Page \& Tab Movements}{}
\subsection{Page Movements}{move commands work as you would expect them to do
from VIM. See {\tt :h motion} for detailed documentation.}
\cmdS{hjkl}	{move left, down, up, right}
\cmdS{gg~G}	{move to top, bottom of page}
\cmdS{\ctrl f~\ctrl d~\ctrl e}	{move screen, page, line down}
\cmdS{\ctrl b~\ctrl u~\ctrl y}	{move screen, page, line up}
\cmdS{/~?}	{search in page, backwards}
\cmdS{n~N}	{jump to next, previous search result}

% subsect %% page_movements (end)

\subsection{Tab \& Buffer Movements}{}
\cmdS{b$n$}	{switch to open buffer $n$ from list}
\cmdS{gt\or\ctrl n}	{go to next tab}
\cmdS{gT\or\ctrl p}	{go to previous tab}
\cmdS{g0~g\$~g@}	{go to first, last, first app tab}
\cmdS{gh~gH}	{open own defined homepage, new tab}
\cmdS{gu~gU}	{move up one path, all to root}
\cmdExmpl{moves up to http\ldots or.org/system}	{gu http://www.vimperator.org/system/app/}
\cmdS{go\lbrack aA-zZ0-9\rbrack}	{jump to previously marked web page}

\section{Basic Commands}{}
\subsection{General Keybinds}	{refer to {\tt :h :open}}
\cmdS{d~u}	{close current, undo last tab}
\cmdS{w~W}	{open new window with empty, current url}
\cmdS{t~T}	{id. but new tab}
\cmdS{o~O}	{id. but in current tab}
\cmdS{y~Y}	{copy url, selected text to clipboard}
\cmdS{p~P}	{paste from clipboard in current, new tab}
\cmdS{H~L}	{go to previous, next page in history}
\cmdS{m~M}	{set local, global mark at cursor, url}
\cmdS{a~A}	{add, toggle bookmark}
\cmdS{i~I}	{ignore any, specific {\tt vimpr} keybind}
\cmdS{\lbracket\lbracket~\rbracket\rbracket}	{switch to previous, next page on multipage websites}
\cmdS{+~-}	{zoom in, out by 25 \% on page in focus}
\cmdS{$n$zz}	{zoom page to $n$\% (100 if $n$ is omitted)}
\cmdS{ZZ~ZQ}	{exit browser, clear history before}

\subsection{Basic Exe commands}{works with tab completion}
\cmdS{:addo\grey{ns}}	{show addons in current window}
\cmdL{:bmark~:delb\grey{marks} $url$}	{add, delete bookmark}
\cmdS{:em\grey{enu} \tab}	{command-line access to Firefox menubar}
\cmdS{:hist\grey{ory}}	{pull up recent 10 online \& local pages}
\cmdS{:pc\grey{lose}}	{close preview window, eg. history}
\cmdS{:pref\grey{erences}}	{show preferences}
\cmdS{:res\grey{tart}}	{restart Firefox immediately}
\cmdS{:tabo\grey{nly}}	{close all other tabs}
\cmdS{:sav\grey{eas}}	{save current web page to disk.}
\cmdL{:se\grey{t} (no)apptab}	{set, unset current tab as app window}

\subsection{QuickHint Mode ($f$)}{Hints are the way in which Vimperator allows you to follow links on a page. By
providing each link with a suitable hint, you can access them via a set of hotkeys.
Pressing $f$ once triggers {\it QuickHint Mode} and simulates a mouseselection with
labeled links on a web page.}
\subsection{ExtendedHint Mode ( ; )}{This mode is useful for
performing operations on hinted elements other than the default
left mouse click. If you want to yank the location of hint 24, press ;y to start this hint
mode. Then press 24 to copy the hint location.}
\subsection{ExtendedHint Commands}{list all extended mode commands {\tt :h extended-hints}}
here are a few \ldots\\
\cmdS{;~}	{imitates mouse-hovering a link object}
\cmdS{s~S}	{save link, object}
\cmdS{i~I}	{open media object (in new tab)}
\cmdS{y~Y}	{copy URL, label description to clipboard}
